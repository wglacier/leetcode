```C++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int ret = 0;
        int max_t = 0;
        vector<int> buf(256, -1);
        for (int i = 0; i < s.size(); i++) {
            char c = s[i];
            if (buf[c] < 0) {
                max_t++;
            } else {
                max_t = min(max_t + 1, i - buf[c]);
            }
            buf[c] = i;
            ret = max(ret, max_t);
        }
        return ret;
    }
};
```
