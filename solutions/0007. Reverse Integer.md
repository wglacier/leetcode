```c++
class Solution {
public:
    int reverse(int x) {
        if (x == INT_MIN) return 0;
        if (x < 0)
            return -1*reverse(-1*x);
        int a = x;
        int res = 0;
        while (a > 0) {
            if (res >= INT_MAX/10 + 1)
                return 0;
            int t = res * 10;
            if (t > INT_MAX - a % 10)
                return 0;

            res = t + a % 10;
            a /= 10;
        }
        return res;
    }
};
```

```C++
class Solution {
public:
    int reverse(int x) {
        int x0 = x;
        if (x < 0) x *= -1;
        
        int r = 0;
        while (x > 0) {
            int t = x % 10;
            if (r*10/10 != r) return 0;
            r = r * 10 + t;
            if (r & 0x80000000) return 0;
            x /= 10;
        }
        if (x0 < 0) r *= -1;
        return r;
    }
};
```
