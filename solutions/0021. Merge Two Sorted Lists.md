### Merge Two Sorted Lists
```c++
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        ListNode dummy(0);
        ListNode* r = &dummy;
        while (l1 != nullptr && l2 != nullptr) {
            if (l1->val < l2->val) {
                r->next = l1;
                r = l1;
                l1 = l1->next;
            } else {
                r->next = l2;
                r = l2;
                l2 = l2->next;
            }
        }
        if (l1) r->next = l1;
        else r->next = l2;
        return dummy.next;
    }
};
```
