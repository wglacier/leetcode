### Merge k Sorted Lists
### using heap
```c++
// 36ms
class Cmp {
public:
	bool operator ()(ListNode* l, ListNode* r) {
		return l->val > r->val;
	}
};

class Solution {
public:
	ListNode* mergeKLists(vector<ListNode*>& lists) {
		priority_queue<ListNode*, vector<ListNode*>,Cmp> qq;
		for (int i = 0; i < lists.size(); i++) {
			ListNode* p = lists[i];
			while (p != NULL) {
				qq.push(p);
				p = p->next;
			}
		}
		if (qq.empty()) return NULL;
		ListNode* t = qq.top();
		ListNode* h = t;
		qq.pop();
		while(!qq.empty()) {
		    t->next = qq.top();
		    qq.pop();
		    t = t->next;
		}
		t->next = NULL;
		return h;
	}
};
```

```c++
class Solution {
public:
    ListNode* mergeTwo(ListNode* l, ListNode* r) {
        ListNode dummy(0);
        ListNode* p = &dummy;
        while( l != nullptr && r != nullptr) {
            if (l->val < r->val) {
                p->next = l;
                p = l;
                l = l->next;
            } else {
                p->next = r;
                p = r;
                r = r->next;
            }
        }
        if (l) p->next = l;
        else if (r) p->next = r;
        return dummy.next;
    }
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        if (lists.empty()) return nullptr;

        while (lists.size() > 1) {
            ListNode* l = lists.back(); lists.pop_back();
            ListNode* r = lists.back(); lists.pop_back();
            ListNode* p = mergeTwo(l, r);
            lists.push_back(p);
        }
        return lists.back();
    }
};
```

```c++
class Solution {
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        ListNode* res = nullptr, *p = nullptr;
        
        while (lists.size() > 1) {
            if (!lists[0]) {
                lists.erase(lists.begin());
                continue;
            }
            int mx = 0;
            for (int i = 1; i < lists.size(); i++) {
                if (!lists[i]) continue;
                if (lists[i]->val < lists[mx]->val)
                    mx = i;
            }
            if (!res) res = p = lists[mx];
            else {
                p->next = lists[mx];
                p = p->next;
            }
            if (!p->next) lists.erase(lists.begin() + mx);
            else lists[mx] = p->next;
        }
        if (!lists.empty()) {
            if (p) p->next = lists[0];
            else return lists[0];
        }
        return res;
    }
    
};
```
