### Length of Last Word
```c++
// 0ms, 100%
class Solution {
public:
    int lengthOfLastWord(string s) {
        if (s.empty()) return 0;
        
        int i = s.size()-1;
        while (i >= 0 && s[i] == ' ')
            i--;
        if (i < 0) return 0;
        
        int j = i;
        while (i >= 0 && s[i] != ' ')
            i--;
        return j - i;
    }
};
```
