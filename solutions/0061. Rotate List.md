### Rotate List
```c++
// 4ms, 100%
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* rotateRight(ListNode* head, int k) {
        if (head == nullptr)
            return head;

        // get the length
        int len = 0;
        ListNode *p = head;
        while (p != nullptr) {
            len++;
            p = p->next;
        }
        k = k % len;
        if (k == 0) return head;
        
        // make p k steps ahead of q
        p = head;
        ListNode *q = head;
        while (k > 0) {
            k--;
            q = q->next;
        }
        // move to the end
        while(q->next != nullptr) {
            p = p->next;
            q = q->next;
        }
        
        q->next = head;
        head = p->next;
        p->next = nullptr;
        return head;
    }
};
```
