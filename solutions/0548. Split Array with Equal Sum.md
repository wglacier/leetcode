# 0548. Split Array with Equal Sum, medium, locked, freq: 4.%, acceptance: 43.7%

Given an array with n integers, you need to find if there are triplets (i, j, k) which satisfies following conditions:

0 < i, i + 1 < j, j + 1 < k < n - 1
Sum of subarrays (0, i - 1), (i + 1, j - 1), (j + 1, k - 1) and (k + 1, n - 1) should be equal.
where we define that subarray (L, R) represents a slice of the original array starting from the element indexed L to the element indexed R.
Example:
Input: [1,2,1,2,1,2,1]
Output: True
Explanation:
i = 1, j = 3, k = 5. 
sum(0, i - 1) = sum(0, 0) = 1
sum(i + 1, j - 1) = sum(2, 2) = 1
sum(j + 1, k - 1) = sum(4, 4) = 1
sum(k + 1, n - 1) = sum(6, 6) = 1
Note:
1 <= n <= 2000.
Elements in the given array will be in range [-1,000,000, 1,000,000].

// 236ms, 18%
class Solution {

public:
    bool splitArray(vector<int>& nums) {
        if (nums.size() < 7)
            return false;
        vector<int> psums;
        psums.reserve(nums.size());
        partial_sum(nums.begin(), nums.end(), back_inserter(psums));
        
        for (int j = 3; j < nums.size()-3; j++) {
            unordered_set<int> ss;
            for (int i = 1; i < j - 1; i++) {
                if (psums[i-1] == psums[j-1] - psums[i])
                    ss.insert(psums[i-1]);
            }
            for (int k = j + 2; k < nums.size() - 1; k++) {
                auto s1 = psums[k-1] - psums[j];
                if (s1 == psums.back() - psums[k] && ss.count(s1) > 0)
                    return true;
            }
        }
        return false;
    }
};

// 636ms, 5%
class Solution {
private:
    bool check(const vector<int>& psums, optional<int>& sum, int s, int e, int N, unordered_map<int, unordered_set<int>>& dp) {
        if (s > e) return false;
        if (sum.has_value()) {
            auto it = dp.find(sum.value());
            if (it != dp.end()) {
                if (it->second.count(s<<16|N) > 0)
                    return false;
            }
        }
        if (N == 1) {
            return psums[e] - psums[s-1] == sum.value();
        }
        for (int i = s + 1; i < e; i++) {
            auto sum0 = sum;
            int sum1 = s > 0? psums[i-1] - psums[s-1] : psums[i-1];
            if (!sum0.has_value())
                sum0 = sum1;
            if (sum1 != sum0.value()) continue;
            if (check(psums, sum0, i+1, e, N-1, dp))
                return true;
        }
        if (sum.has_value()) {
            dp[sum.value()].insert(s<<16|N);
        }
        return false;
    }
public:
    bool splitArray(vector<int>& nums) {
        vector<int> psums;
        psums.reserve(nums.size());
        partial_sum(nums.begin(), nums.end(), back_inserter(psums));
        optional<int> sum;
        unordered_map<int, unordered_set<int>> dp;
        return check(psums, sum, 0, nums.size()-1, 4, dp);
    }
};
