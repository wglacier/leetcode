        vector<pair<TreeNode*, int>> q;
        q.emplace_back(root, 1);
        int res = 0;
        while (!q.empty()) {
            decltype(q) q2;
            int i = -1, j = -1, base = -1;
            for (const auto& rr : q) {
                auto r = rr.first;
                auto c = rr.second;
                if (base < 0) {
                    base = c;
                }
                c -= base;
                if (i < 0) {
                    i = c;
                }
                j = c;
                if (r->left) {
                    q2.emplace_back(r->left, c * 2);
                }
                if (r->right) {
                    q2.emplace_back(r->right, c * 2 + 1);
                }
            }
            res = max(res, j - i + 1);
            q.swap(q2);
        }
        return res;
    }
