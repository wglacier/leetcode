### Swap Nodes in Pairs
```c++
class Solution {
public:
    ListNode* swapPairs(ListNode* head) {
        ListNode dummy(0);
        ListNode* t = &dummy;
        t->next = head;
        while (t) {
            ListNode* p = t->next;
            if (!p || !p->next) break;
            
            ListNode* q = p->next;
            ListNode* r = q->next;
            
            t->next = q;
            q->next = p;
            p->next = r;
            t = p;
        }
        return dummy.next;
    }
};

// recursive way
ListNode* swapPairs(ListNode* head) {
    if (!head || !head->next) return head;
    
    ListNode* p = head->next;
    ListNode* q = p->next;
    p->next = head;
    head->next = swapPairs(q);
    return p;
}
```
