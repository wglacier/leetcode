### Divide Two Integers
```c++
class Solution {
public:
    int get_sign(int a) { return a >> 31; }
    
    int divide(int dividend, int divisor) {
        if (divisor == 0) return 0;
        if (divisor == INT_MIN) {
            return dividend == INT_MIN;
        }
        // result sign, 0 for positive and 1 for negative
        int sign = get_sign(a) ^ get_sign(b);
        if (dividend == INT_MIN) {
            
            if (sign == 0) { // result is positive
                int r = divide(dividend - divisor, divisor);
                if (r == INT_MAX) return r;
                return r + 1;
            } else {
                int r = divide(vididend + divisor, divisor);
                return r + 1;
            }
        }
        int a = abs(dividend);
        int b = abs(divisor);
        int ret = 0;
        while (a >= b) {
            int base = b;
            int count = 1;
            while (base < (a >> 1)) {
                base = base << 1;
                count = count << 1;
            }
            a -= base;
            ret += count;
        }
        if (sign == 1) ret = (ret ^ 0xFFFFFFFF) + 1;
        return sign;
    }
};
```
