```C++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        if (s.empty()) return 0;
        
        vector<int> buf(256, -1);
        int ans = 0, t = 0;
        for (int i = 0; i < s.size(); i++) {
            char c = s[i];
            if (buf[c] >= 0) {
                ans = max(ans, t);
                t = min(i - buf[c], t + 1);
            } else {
                t += 1;
            }
            buf[c] = i;
        }
        ans = max(ans, t);
        return ans;
    }
};
```
