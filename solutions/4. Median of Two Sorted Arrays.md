class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
        int z = nums1.size() + nums2.size();
        int m = (z -1) / 2;
        int i = 0, j = 0;
        int k = 0;
        int v1 = 0, v2 = 0;
        while (i < nums1.size() || j < nums2.size()) {
            int v = 0;
            if (j >= nums2.size() || ( i < nums1.size() && nums1[i] < nums2[j])) {
                v = nums1[i];
                i += 1;
            } else {
                v = nums2[j];
                j += 1;
            }
            if (k == m) {
                v1 = v;
            } else if (k == m+1) {
                v2 = v;
                break;
            }
            k += 1;
        }
        if (z % 2 != 0) {
            return v1;
        } else {
            return (v1 + v2) / 2.0;
        }
    }
};
